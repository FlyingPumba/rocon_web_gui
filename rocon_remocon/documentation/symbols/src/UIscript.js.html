 77 /**
 78   * Read cookies and add to url list
 79   *
 80   * @function readCookies
 81 */
 82 function readCookies() {
 83   $.cookie.defaults = { path: '/', expires: 365 };
 84 
 85   gCookieCount = $.cookie("cookieCount");
 86 
 87   // init cookie count
 88   if (gCookieCount == null || gCookieCount < 0) {
 89     $.cookie("cookieCount", 0);
 90     gCookieCount = 0;
 91   }
 92 
 93   // read cookie and add to url list
 94   for (var i = 0; i < gCookieCount; i++) {
 95     $("#urlList").append(new Option($.cookie("cookie_url" + i)));
 96     $("#urlList option:last").attr("cookieNum", i);
 97   }
 98 }

100 /**
101   * Event function when 'Connect' button clicked
102   *
103   * @function connect
104 */
105 function connect() {
106   $("#connectBtn").click(function () {
107     var url = $("#urlList option:selected").val();
108     
109     if (url == "(Please add URL)") {
110       return;
111     }
112 
113     gUrl = url;
114 
115     // extract the exact url
116     var newUrl;
117     newUrl = url.replace("ws://", "");
118 
119     for (var i = 0; i < newUrl.length; i++) {
120       newUrl = newUrl.replace("/", "");
121       newUrl = newUrl.replace(" ", "");
122     }
123         
124     // set default port
125     if (newUrl.search(":") < 0) {
126       newUrl += ":9090";
127     }
128 
129     try {
130       ros.connect('ws://' + newUrl);
131       $(this).text = "Disconnect";
132     } catch(e) {
133       console.log(e.message);
134       alert(e.message);
135     }
136   });
137 }

177 /**
178   * Event function when 'Minus' button clicked
179   *
180   * @function deleteUrl
181 */
182 function deleteUrl() {
183   $("#urldeleteBtn").click(function () {
184     if ($("#urlList option:selected").val() != "(Please add URL)") {
185       // delete cookie
186       var cookieNum = $("#urlList option:selected").attr("cookieNum");
187       $.cookie("cookie_url" + cookieNum, null);
188             
189       if (gCookieCount > 0) {
190         $.cookie("cookieCount", --gCookieCount);
191       }
192             
193       $("#urlList option:selected").remove();
194 
195       var listCount = $("#urlList option").length;
196       var tempCount = 0;
197 
198       // rearrange cookies
199       // not including the first disabled option
200       for (var i = 1; i < listCount; i++) {
201         var url = $("#urlList option:eq(" + i + ")").val();
202 
203         $("#urlList option:eq(" + i + ")").attr("cookieNum", tempCount);
204         $.cookie("cookie_url" + tempCount, url);
205         tempCount++;
206       }
207     }
208   });
209 }

293 /**
294   * Classify the interaction whether it's (web_url) or (web_app)
295   *
296   * @function classifyInteraction
297   *
298   * @param {interaction} interaction
299   * @returns {string} extracted url
300 */
301 function classifyInteraction(interaction) {
302   var newUrl;
303   var url = interaction.name;
304 
305   if (url.search("web_url") >= 0) {
306     newUrl = url.substring(8, url.length - 1);
307   }
308   else if (url.search("web_app") >= 0) {
309     var tempUrl = url.substring(8, url.length - 1);
310     newUrl = prepareWebappUrl(interaction, tempUrl);
311   }
312   else {
313     newUrl = null;
314   }
315 
316   return newUrl;
317 }
318 
319 /**
320   * Url synthesiser for sending remappings and parameters information
321   *
322   * @function prepareWebappUrl
323   * 
324   * @param {interaction} interaction
325   * @param {string} baseUrl - url before edited
326   * @returns {string} the final remapped url
327 */
328 function prepareWebappUrl(interaction, baseUrl) {
329   // convert and set the informations
330   var interactionData = {};
331   interactionData['display_name'] = interaction.display_name;
332   interactionData['parameters'] = jsyaml.load(interaction.parameters);
333   interactionData['remappings'] = {};
334 
335   $.each(interaction.remappings, function(key, value) {
336     interactionData['remappings'][value.remap_from] = value.remap_to;
337   });
338     
339   // Package all the data in json format and dump it to one query string variable
340   queryStringMappings = {};
341   queryStringMappings['interaction_data'] = JSON.stringify(interactionData);
342     
343   // Encode the url and finish constructing
344   var url = baseUrl + "?interaction_data=" + encodeURIComponent(queryStringMappings['interaction_data']);
345 
346   return url;
347 }

511 /**
512   * Wrapper function for Service.callService
513   *
514   * @function callService
515   *
516   * @param {ROSLIB.Ros} ros - handled ros
517   * @param {string} serviceName - service's name
518   * @param {string} serviceType - service's type
519   * @param {ROSLIB.ServiceRequest} request - request
520   * @param {callBack} callback for request response
521 */
522 function callService(ros, serviceName, serviceType, request, callBack) {
523   var service = new ROSLIB.Service({
524     ros : ros,
525     name : serviceName,
526     serviceType : serviceType
527   });
528 
529   // get response
530   try {
531     service.callService(request, function(result){
532     callBack(result);
533     }, 
534     function(error) {
535       alert(error);
536       console.log(error);
537     });
538   } catch (e) {
539       console.log(message);
540       alert(e.message);
541   } 
542 }

544 /**
545   * Wrapper function for Topic.subscribe
546   *
547   * @function subscribeTopic
548   *
549   * @param {ROSLIB.Ros} ros - handled ros
550   * @param {string} topicName - topic's name
551   * @param {string} msgType - message type
552   * @param {callBack} callback for returned message
553 */
554 function subscribeTopic(ros, topicName, msgType, callBack) {
555   var listener = new ROSLIB.Topic({
556     ros : ros,
557     name : topicName,
558     messageType : msgType
559   });
560     
561   // get returned message
562   listener.subscribe(function(message) {
563     callBack(message);
564     listener.unsubscribe();
565   });
566 }

568 /**
569   * Get browser name
570   *
571   * @function getBrowser
572   *
573   * @returns {string} current browser's name
574 */
575 function getBrowser() {
576   var agt = navigator.userAgent.toLowerCase();
577   if (agt.indexOf("chrome") != -1) return 'chrome';
578   if (agt.indexOf("opera") != -1) return 'opera';
579   if (agt.indexOf("firefox") != -1) return 'firefox';
580   if (agt.indexOf("safari") != -1) return 'safari';
581   if (agt.indexOf("msie") != -1) return 'internet_explorer';
582 }
